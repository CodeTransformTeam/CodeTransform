  7. public class HTMLWriter {

  15.   void write(File file) { 
  16.     try { 
  17.       if (file.exists() == false) { 
  18.         file.createNewFile(); 
  19.       } 
  20.  
  21.       FileOutputStream fileOutputStream = new FileOutputStream(file); 
  22.       writeHeader(fileOutputStream, file.getName()); 
  23.       writeBody(fileOutputStream); 
  24.       writeFooter(fileOutputStream); 
  25.       fileOutputStream.close(); 
  26.     } catch (Exception e) { 
  27.       e.printStackTrace(); 
  28.     } 
  29.  
  30.   } 
  31.  
  32.   private void writeFooter(FileOutputStream fileOutputStream) { 
  33.     String footerString = "</body>\r\n</html>"; 
  34.     try { 
  35.       fileOutputStream.write(footerString.getBytes()); 
  36.     } catch (Exception e) { 
  37.       e.printStackTrace(); 
  38.     } 
  39.   } 
  40.  
  41.   private String replaceEscapeChars(String src) { 
  42.     src = src.replace("&", "&amp;"); 
  43.     src = src.replace(" ", "&nbsp;"); 
  44.     src = src.replace("\r\n", "&nbsp;\r\n"); 
  45.     src = src.replace("\t", "&nbsp;&nbsp;"); 
  46.     src = src.replace("<", "&lt;"); 
  47.     src = src.replace(">", "&gt;"); 
  48.  
  49.     return src; 
  50.   } 
  51.  
  52.   private void writeBody(FileOutputStream fileOutputStream) { 
  53.     ArrayList<ParsedCode> parsedCodes = parser_.getParserResult(); 
  54.     try { 
  55.       fileOutputStream 
  56.           .write("  <div class=\"code\" style=\"background-color:#E7E5DC\">\r\n" 
  57.               .getBytes()); 
  58.       fileOutputStream 
  59.           .write("    <ol start=\"1\" style=\"margin:0 0 1px 45px !important;\">\r\n" 
  60.               .getBytes()); 
  61.  
  62.       for (int i = 0; i < parsedCodes.size(); i++) { 
  63.         ParsedCode parsedCode = parsedCodes.get(i); 
  64.         fileOutputStream.write("      <li>\r\n".getBytes()); 
  65.         String colorString = ColorConverter 
  66.             .Color2String(parsedCode.codeColor_); 
  67.         String fontString = parsedCode.codeFont_; 
  68.         String spanString = "        <span style=\"color:" + colorString 
  69.             + "; font:" + fontString + "\" >"; 
  70.         fileOutputStream.write(spanString.getBytes()); 
  71.         do { 
  72.           String codeString = parsedCode.codeString_; 
  73.            
  74.           //检测换行个数，两个以上的话拆分成开 
  75.           int index = codeString.indexOf("\n"); 
  76.           if (index >= 0) { 
  77.             index++; 
  78.             if (index < codeString.length()) { 
  79.               // 至少有两个换行 
  80.               String rightString = codeString.substring(index); 
  81.               ParsedCode parsedCodeNew = new ParsedCode(); 
  82.               parsedCodeNew.codeColor_ = parsedCode.codeColor_; 
  83.               parsedCodeNew.codeString_ = rightString; 
  84.               parsedCodeNew.codeFont_ = parsedCode.codeFont_; 
  85.               parsedCodes.add(i+1, parsedCodeNew); 
  86.                
  87.               String leftString = codeString.substring(0, index); 
  88.               parsedCode.codeString_ = leftString; 
  89.               parsedCodes.set(i, parsedCode); 
  90.                
  91.               codeString = leftString; 
  92.             } 
  93.           } 
  94.            
  95.           codeString = replaceEscapeChars(codeString); 
  96.           fileOutputStream.write(codeString.getBytes()); 
  97.  
  98.           if (codeString.indexOf("\n") > 0) { 
  99.             break; 
  100.           } 
  101.  
  102.           if (i < parsedCodes.size() - 1) { 
  103.             ParsedCode nextCode = parsedCodes.get(i + 1); 
  104.             if (nextCode.codeColor_.equals(parsedCode.codeColor_) == false 
  105.                 && nextCode.codeString_.indexOf('\n') < 0) { 
  106.               // 颜色不一样，又不是回车 
  107.               colorString = ColorConverter 
  108.                   .Color2String(nextCode.codeColor_); 
  109.               fontString = nextCode.codeFont_; 
  110.               spanString = "</span><span style=\"color:" + colorString 
  111.                   + "; font:" + fontString + "\" >"; 
  112.               fileOutputStream.write(spanString.getBytes()); 
  113.  
  114.               parsedCode = nextCode; 
  115.               ++i; 
  116.             } else if (nextCode.codeColor_ 
  117.                 .equals(parsedCode.codeColor_)) { 
  118.               parsedCode = nextCode; 
  119.               ++i; 
  120.             } else { 
  121.               parsedCode = nextCode; 
  122.               ++i; 
  123.             } 
  124.           } else { 
  125.             break; 
  126.           } 
  127.         } while (i < parsedCodes.size()); 
  128.         spanString = "        </span>\r\n"; 
  129.         fileOutputStream.write(spanString.getBytes()); 
  130.  
  131.         fileOutputStream.write("      </li>\r\n".getBytes()); 
  132.       } 
  133.  
  134.       fileOutputStream.write("    </ol>\r\n".getBytes()); 
  135.       fileOutputStream.write("  </div>\r\n".getBytes()); 
  136.     } catch (Exception e) { 
  137.       e.printStackTrace(); 
  138.     } 
  139.   }

  141.   private void writeHeader(FileOutputStream fileOutputStream, String title) { 
  142.     String headerString = "<!DOCTYPE html PUBLIC \"-//W3C//DTD HTML 4.01 Transitional//EN\" " 
  143.         + "\"http://www.w3.org/TR/html4/loose.dtd\">\r\n" 
  144.         + "<html>\r\n" 
  145.         + "<head>\r\n" 
  146.         + "<meta http-equiv=\"Content-Type\" content=\"text/html; charset=UTF-8\">\r\n" 
  147.         + "<title>" 
  148.         + title 
  149.         + "</title>\r\n" 
  150.         + "</head>\r\n" 
  151.         + "<body>\r\n"; 
  152.     try { 
  153.       fileOutputStream.write(headerString.getBytes()); 
  154.       System.out.println("HTMLWriter.writeHeader()"); 
  155.     } catch (Exception e) { 
  156.       e.printStackTrace(); 
  157.     } 
  158.   }
