  89.     private boolean isCharPrintable(byte ch) { 
  90.         if (ch > 32 && ch < 127) { 
  91.             return true; 
  92.         } 
  93.          
  94.         return false; 
  95.     } 
  96.      
  97.     /* 
  98.      * 这个方法，将buffer里面的单词分开， 
  99.      * 可打印的一堆，不可打印的一堆 
  100.      * 例如 "abc+-\r\ndef\t\tgh"分成 
  101.      * "abc+-"、"\r\n"、"\t\t"、"gh" 
  102.      */ 
  103.     protected ArrayList<String> parsePrintableWords(byte[] buffer) { 
  104.         ArrayList<String> result = new ArrayList<String>(); 
  105.         Boolean stringPrintable = false; 
  106.          
  107.         byte[] splitBuffer = new byte[buffer.length]; 
  108.         int j = 0; 
  109.          
  110.         for (int i = 0; i < buffer.length; i++) { 
  111.             boolean charPrintable = isCharPrintable(buffer[i]); 
  112.             if (charPrintable != stringPrintable) { 
  113.                 stringPrintable = charPrintable; 
  114.                 if (j > 0) { 
  115.                     String string = new String(splitBuffer, 0, j); 
  116.                     result.add(string); 
  117.                     //清空掉 
  118.                     j = 0; 
  119.                 } 
  120.             }  
  121.             splitBuffer[j++] = buffer[i]; 
  122.         } 
  123.          
  124.         if (j > 0) { 
  125.             String string = new String(splitBuffer, 0, j); 
  126.             result.add(string); 
  127.         } 
  128.          
  129.         return result; 
  130.     }
